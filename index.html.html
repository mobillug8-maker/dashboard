Apps Script 설정
스프레드시트에서 확장 프로그램 → Apps Script 클릭 후, 아래 코드를 붙여넣기 하세요.
function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('실적데이터'); var data = JSON.parse(e.postData.contents); var rows = sheet.getDataRange().getValues(); // 동일 연월+일 행 찾아서 업데이트 또는 추가 var found = false; for (var i = 1; i < rows.length; i++) { if (rows[i][0] == data.연월 && rows[i][1] == data.일) { sheet.getRange(i+1, 1, 1, 6).setValues( [[data.연월, data.일, data.견적, data.계약, data.취소, data.출고]]); found = true; break; } } if (!found) { sheet.appendRow([data.연월, data.일, data.견적, data.계약, data.취소, data.출고]); } return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('실적데이터'); var rows = sheet.getDataRange().getValues(); var data = []; for (var i = 1; i < rows.length; i++) { if (rows[i][0]) { data.push({ 연월: rows[i][0], 일: rows[i][1], 견적: rows[i][2], 계약: rows[i][3], 취소: rows[i][4], 출고: rows[i][5] }); } } return ContentService .createTextOutput(JSON.stringify({data:data})) .setMimeType(ContentService.MimeType.JSON); }
